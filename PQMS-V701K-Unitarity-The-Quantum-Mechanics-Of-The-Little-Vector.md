    def calculate_fidelity(self, initial_state: np.ndarray, evolved_state: np.ndarray) -> float:
        """
        Calculates the fidelity between an initial quantum state and an evolved state.
        Fidelity is a measure of how similar two quantum states are. For pure states,
        it's the squared overlap.

        Args:
            initial_state (np.ndarray): The initial quantum state vector.
            evolved_state (np.ndarray): The evolved (or target) quantum state vector.

        Returns:
            float: The fidelity value, between 0.0 (completely different) and 1.0 (identical).
        """
        if initial_state.shape != evolved_state.shape:
            self.log_error("State vectors must have the same shape for fidelity calculation.")
            raise ValueError("State vector shape mismatch.")

        # Ensure states are normalized
        norm_initial = np.linalg.norm(initial_state)
        norm_evolved = np.linalg.norm(evolved_state)
        if not np.isclose(norm_initial, 1.0) or not np.isclose(norm_evolved, 1.0):
            self.log_warning("Input states for fidelity calculation are not normalized. Normalizing internally.")
            initial_state = initial_state / norm_initial
            evolved_state = evolved_state / norm_evolved

        fidelity = np.abs(np.vdot(initial_state, evolved_state))**2
        self._current_fidelity = fidelity
        self.log_debug(f"Calculated RCF: {fidelity:.12f}.")
        return fidelity

    @property
    def current_fidelity(self) -> float:
        """Returns the last calculated or current estimated coherence fidelity."""
        return self._current_fidelity
