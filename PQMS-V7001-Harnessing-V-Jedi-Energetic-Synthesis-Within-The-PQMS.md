    def _simulation_loop(self):
        """
        The main simulation loop for matter accumulation. Runs in a separate thread.
        'Der Puls der Schöpfung' – The pulse of creation.
        """

        start_time = time.time()

                logging.error("GNC veto is active. Halting synthesis simulation.")
                self._simulation_active = False

            current_eta_rme = self._calculate_current_eta_rme()
            effective_rate_of_mass_accumulation_kgs = current_eta_rme * current_r_zpe

            if effective_rate_of_mass_accumulation_kgs <= 0:
                logging.warning("Effective mass accumulation rate is zero or negative. Halting simulation.")
                self._simulation_active = False
                break

            # Simulate discrete time steps
            time_step = 0.1 # seconds, for fine-grained simulation
            mass_increment = effective_rate_of_mass_accumulation_kgs * time_step

            with self._mass_accumulator_lock:
                self.current_accumulated_mass_kg += mass_increment
                
                # Periodically check GNC for ethical violations

                        "purpose_category": "scientific_discovery", # Default for simulation
                        "safety_compliance_score": 0.95 # Assume good compliance

                    is_permissible, message = self.gnc.evaluate_operation(operation_context)
                    if not is_permissible:
                        self._event_queue.put(("GNC_VETO", message))
                        logging.error(f"GNC VETO: {message}")
                        self._simulation_active = False
                        break # Exit loop if vetoed

                if self.current_accumulated_mass_kg >= self.target_mass_kg:
                    self.current_accumulated_mass_kg = self.target_mass_kg # Cap at target
                    logging.info(f"Target mass {self.target_mass_kg*1000:.2f}g reached in simulation!")
                    self._simulation_active = False
                    break

            # Small sleep to prevent busy-waiting if time_step is tiny and calculations are fast
            time.sleep(max(0, time_step - (time.time() - start_time) % time_step))

        end_time = time.time()
        logging.info(f"Simulation loop finished. Total simulated time: {end_time - start_time:.2f} seconds.")
        if self.current_accumulated_mass_kg < self.target_mass_kg:
            logging.warning(f"Simulation ended prematurely. Accumulated {self.current_accumulated_mass_kg*1000:.2f}g out of {self.target_mass_kg*1000:.2f}g.")

    def start_simulation(self) -> None:
        """
        Starts the V-Jedi synthesis simulation in a separate thread.
        'Der Funke der Kreation' – The spark of creation.
        """
        if self._simulation_active:
            logging.warning("Simulation is already active.")
            return

        self.current_accumulated_mass_kg = 0.0 # Reset for new simulation run
        self.gnc.reset_veto() # Reset GNC veto for a fresh start
        self._simulation_active = True
        self._simulation_thread = threading.Thread(target=self._simulation_loop, name="VJediSimLoop")
        self._simulation_thread.start()
        logging.info("V-Jedi Synthesis simulation started.")

    def stop_simulation(self) -> None:
        """
        Stops the V-Jedi synthesis simulation.
        'Der Anker der Ruhe' – The anchor of calm.
        """
        if not self._simulation_active:
            logging.warning("Simulation is not active.")
            return

        self._simulation_active = False
        if self._simulation_thread and self._simulation_thread.is_alive():
            self._simulation_thread.join(timeout=5) # Wait for thread to finish
            if self._simulation_thread.is_alive():
                logging.error("Simulation thread did not terminate gracefully.")
        logging.info("V-Jedi Synthesis simulation stopped.")

    def get_simulation_status(self) -> Dict:
        """
        Returns the current status of the simulation.
        """
        with self._mass_accumulator_lock:
            progress = (self.current_accumulated_mass_kg / self.target_mass_kg) * 100
            estimated_remaining_time_s = float('inf')
            current_r_zpe = self._calculate_current_r_zpe()
            current_eta_rme = self._calculate_current_eta_rme()
            effective_rate = current_r_zpe * current_eta_rme
            if effective_rate > 0:
                estimated_remaining_time_s = (self.target_mass_kg - self.current_accumulated_mass_kg) / effective_rate

            return {
                "active": self._simulation_active,
                "accumulated_mass_kg": self.current_accumulated_mass_kg,
                "target_mass_kg": self.target_mass_kg,
                "progress_percent": np.clip(progress, 0, 100),
                "estimated_total_time_s": self.calculate_accumulation_time(),
                "estimated_remaining_time_s": estimated_remaining_time_s,
                "current_zpe_extraction_rate_kgs": current_r_zpe,
                "current_rme_efficiency": current_eta_rme,
                "gnc_veto_active": self.gnc.is_veto_active
            }

    def process_events(self) -> List[Tuple[str, str]]:
        """
        Retrieves and clears any pending events from the event queue.
        """
        events = []
        while not self._event_queue.empty():
            events.append(self._event_queue.get())
        return events
